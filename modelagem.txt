product:
    product_id (Primary Key)
    created_at
    title
    price
    img_path

user:
    user_id (Primary Key)
    created_at
    name
    email
    password (hashed and salted)
    role (ENUM)

cart:
    cart_id (Primary Key)
    created_at
    user_id (Foreign Key, links to User)

cart_product:
    cart_product_id (Primary Key)
    created_at
    cart_id (Foreign Key, links to Cart)
    product_id (Foreign Key, links to Product)
    quantity

payment_method:
    payment_method_id (Primary Key)
    created_at
    name (e.g., Credit Card, Debit Card)

order:
    order_id (Primary Key)
    created_at
    user_id (Foreign Key, links to User)

order_item:
    order_item_id (Primary Key)
    created_at
    order_id (Foreign Key, links to Order)
    product_id (Foreign Key, links to Product)
    quantity
